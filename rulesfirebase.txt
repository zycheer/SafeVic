        async function loadRescuersData() {
            console.log('🔄 Loading rescuers data from API...');
            
            const loadingState = document.getElementById('loadingState');
            const errorState = document.getElementById('errorState');
            const refreshBtn = document.getElementById('refreshBtn');
            const refreshIcon = refreshBtn?.querySelector('[data-feather="refresh-cw"]');
            
            try {
                // Show loading state
                if (loadingState) {
                    loadingState.classList.remove('hidden');
                }
                
                // Hide error state
                if (errorState) {
                    errorState.classList.add('hidden');
                }
                
                // Disable refresh button
                if (refreshBtn) {
                    refreshBtn.disabled = true;
                    if (refreshIcon) {
                        refreshIcon.style.animation = 'spin 1s linear infinite';
                    }
                }
                
                // Fetch data from your API
                const response = await fetch('/api/rescuers', {
                    method: 'GET',
                    headers: {
                        'Content-Type': 'application/json'
                    }
                });
                
                if (!response.ok) {
                    throw new Error(`HTTP error! status: ${response.status}`);
                }
                
                const data = await response.json();
                
                if (data.success) {
                    const allowedRoles = ['BFP', 'MDRRMO', 'PNP', 'MEDICAL'];
                    rescuersData = (data.rescuers || []).filter(r =>
                    allowedRoles.includes((r.type || '').toUpperCase())
                    );
                    filteredData = [...rescuersData];

                    
                    // CRITICAL FIX: Store rescuers by index for safe access
                    rescuersByIndex.clear();
                    rescuersData.forEach((rescuer, index) => {
                        rescuersByIndex.set(index, rescuer);
                    });
                    
                    console.log('✅ Data loaded:', {
                        totalRescuers: rescuersData.length,
                        sampleRescuer: rescuersData[0] || 'No rescuers'
                    });
                    
                    loadRescuersTable();
                    updateRescuerStats();
                    updateRescuerCount();
                    
                    showNotification(`Successfully loaded ${rescuersData.length} rescuers`, 'success');
                } else {
                    throw new Error(data.error || 'Failed to load rescuers');
                }
                
                // Hide loading state
                if (loadingState) {
                    loadingState.classList.add('hidden');
                }
                
                // Re-enable refresh button
                if (refreshBtn) {
                    refreshBtn.disabled = false;
                    if (refreshIcon) {
                        refreshIcon.style.animation = '';
                    }
                }
                
                if (typeof feather !== 'undefined') {
                    feather.replace();
                }
                
            } catch (error) {
                console.error('❌ Error loading rescuers:', error);
                
                if (errorState) {
                    const errorMessage = document.getElementById('errorMessage');
                    if (errorMessage) {
                        errorMessage.textContent = `Error: ${error.message}`;
                    }
                    errorState.classList.remove('hidden');
                }
                
                if (loadingState) {
                    loadingState.classList.add('hidden');
                }
                
                if (refreshBtn) {
                    refreshBtn.disabled = false;
                    if (refreshIcon) {
                        refreshIcon.style.animation = '';
                    }
                }
                
                showNotification(`Failed to load rescuers: ${error.message}`, 'error');
            }
        }

                    element.value = value;

                        const options = Array.from(element.options);
                        const match = options.find(opt => 
                            opt.value.toLowerCase() === value.toLowerCase() ||

                        );

// Enhanced location API functions
async function populateLocationDropdowns(rescuer) {
    try {
        console.log('🌍 Loading location data for rescuer:', rescuer.address);
        
        // Load provinces first
        await loadProvinces();
        
        // If rescuer has address data, populate dropdowns in sequence
        if (rescuer.address) {
            const { province, municipality, barangay } = rescuer.address;
            
            if (province) {
                // Set province
                const provinceSelect = document.getElementById('edit_province');
                if (provinceSelect) {
                    provinceSelect.value = province;
                    
                    // Load municipalities for this province
                    if (municipality) {
                        await loadMunicipalities(province, municipality);
                        
                        // Load barangays for this municipality
                        if (barangay) {
                            await loadBarangays(municipality, barangay);
                        }
                    }
                }
            }
        }
    } catch (error) {
        console.error('Error loading location data:', error);
        showNotification('Error loading location data', 'error');
    }
}

        throw error;
    }
}

// Enhanced municipality loading
async function loadMunicipalities(provinceName, selectedMunicipality = null) {
    const municipalitySelect = document.getElementById('edit_municipality');
    const barangaySelect = document.getElementById('edit_barangay');
    
    if (!municipalitySelect) return;
    
    try {
        municipalitySelect.disabled = true;
        municipalitySelect.innerHTML = '<option value="">Loading...</option>';
        
        // Reset barangay dropdown
        if (barangaySelect) {
            barangaySelect.innerHTML = '<option value="">Select Barangay</option>';
            barangaySelect.disabled = true;
        }
        
        // Get province code from the selected option
        const provinceSelect = document.getElementById('edit_province');
        const selectedProvinceOption = provinceSelect.options[provinceSelect.selectedIndex];
        const provinceCode = selectedProvinceOption?.getAttribute('data-code');
        
        if (!provinceCode && !provinceName) {
            throw new Error('Province code or name required');
        }
        
        // Use either province code or name for the API call
        const endpoint = provinceCode ? 
            `/api/locations/provinces/${provinceCode}/municipalities` :
            `/api/locations/municipalities?province=${encodeURIComponent(provinceName)}`;
            
        const response = await fetch(endpoint);
        if (!response.ok) throw new Error('Failed to load municipalities');
        
        const municipalities = await response.json();
        
        municipalitySelect.innerHTML = '<option value="">Select Municipality</option>';
        municipalities.forEach(municipality => {
            const option = document.createElement('option');
            option.value = municipality.name;
            option.setAttribute('data-code', municipality.code);
            option.textContent = municipality.name;
            municipalitySelect.appendChild(option);
        });
        
        municipalitySelect.disabled = false;
        
        // If there's a selected municipality, set it
        if (selectedMunicipality) {
            municipalitySelect.value = selectedMunicipality;
        }
        
        console.log('✅ Municipalities loaded successfully');
    } catch (error) {
        console.error('Error loading municipalities:', error);
        municipalitySelect.innerHTML = '<option value="">Error loading</option>';
        municipalitySelect.disabled = false;
        throw error;
    }
}

// Updated form submission handler
async function handleRescuerFormSubmit(event) {
    event.preventDefault();
    
    console.log('Form submission started', { isAddingNew });
    
    const submitBtn = document.getElementById('saveRescuerBtn');
    const originalText = submitBtn.innerHTML;
    
    try {
        submitBtn.disabled = true;
        submitBtn.innerHTML = '<i data-feather="loader" class="w-4 h-4 animate-spin"></i> Processing...';
        
        const formData = new FormData(event.target);
        
        if (isAddingNew) {
            // Registration mode - validate passwords
            const password = formData.get('password');
            const confirmPassword = formData.get('confirmPassword');
            
            if (!password || password.length < 6) {
                throw new Error('Password must be at least 6 characters long');
            }
            
            if (password !== confirmPassword) {
                throw new Error('Passwords do not match');
            }
            
            // Additional validation for registration
            const validationErrors = validateRegistrationForm(formData);
            if (validationErrors.length > 0) {
                throw new Error(validationErrors.join(', '));
            }
            
            // Prepare registration data
            const registrationData = {
                email: formData.get('email').trim(),
                password: password,
                firstName: formData.get('firstName').trim(),
                middleInitial: formData.get('middleInitial').trim(),
                lastName: formData.get('lastName').trim(),
                phoneNumber: formData.get('phoneNumber').trim(),
                gender: formData.get('gender'),
                role: formData.get('role'),
                address: {
                    province: formData.get('province'),
                    municipality: formData.get('municipality'),
                    barangay: formData.get('barangay'),
                    street: formData.get('street').trim()
                }
            };
            
            // Add codes if available
            const provinceSelect = document.getElementById('edit_province');
            const municipalitySelect = document.getElementById('edit_municipality');
            const barangaySelect = document.getElementById('edit_barangay');
            
            if (provinceSelect.selectedIndex > 0) {
                registrationData.address.provinceCode = provinceSelect.options[provinceSelect.selectedIndex].getAttribute('data-code');
            }
            if (municipalitySelect.selectedIndex > 0) {
                registrationData.address.municipalityCode = municipalitySelect.options[municipalitySelect.selectedIndex].getAttribute('data-code');
            }
            if (barangaySelect.selectedIndex > 0) {
                registrationData.address.barangayCode = barangaySelect.options[barangaySelect.selectedIndex].getAttribute('data-code');
            }
            
            console.log('Registering new rescuer:', registrationData);
            
            // Call the new admin registration API endpoint
            const response = await fetch('/api/admin/register-rescuer', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify(registrationData)
            });
            
            if (!response.ok) {
                const errorText = await response.text();
                let errorMessage = 'Registration failed';
                
                try {
                    const errorData = JSON.parse(errorText);
                    errorMessage = errorData.error || errorMessage;
                } catch {
                    errorMessage = errorText || errorMessage;
                }
                
                throw new Error(errorMessage);
            }
            
            const result = await response.json();
            
            if (result.success) {
                showNotification('Rescuer registered successfully!', 'success');
                closeEditModal();
                loadRescuersData();
            } else {
                throw new Error(result.error || 'Registration failed');
            }
            
        } else {
            // Edit mode - existing update functionality
            if (!currentEditingRescuer || !currentEditingRescuer.id) {
                throw new Error('No rescuer selected for editing');
            }
            
            // Prepare update data (without password fields)
            const updateData = {
                firstName: formData.get('firstName').trim(),
                middleInitial: formData.get('middleInitial').trim(),
                lastName: formData.get('lastName').trim(),
                phoneNumber: formData.get('phoneNumber').trim(),
                gender: formData.get('gender'),
                role: formData.get('role'),
                emergencyContact: formData.get('emergencyContact').trim(),
                address: {
                    province: formData.get('province'),
                    municipality: formData.get('municipality'),
                    barangay: formData.get('barangay'),
                    street: formData.get('street').trim()
                }
            };
            
            const response = await fetch(`/api/rescuers/${currentEditingRescuer.id}/update`, {
                method: 'PUT',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify(updateData)
            });
            
            if (!response.ok) {
                const errorText = await response.text();
                throw new Error(`Update failed: ${errorText}`);
            }
            
            const result = await response.json();
            
            if (result.success) {
                showNotification('Rescuer updated successfully!', 'success');
                closeEditModal();
                loadRescuersData();
            } else {
                throw new Error(result.error || 'Update failed');
            }
        }
        
    } catch (error) {
        console.error('Error saving rescuer:', error);
        showNotification(error.message, 'error');
    } finally {
        submitBtn.disabled = false;
        submitBtn.innerHTML = originalText;
        if (typeof feather !== 'undefined') feather.replace();
    }
}

// Validation for registration form
function validateRegistrationForm(formData) {
    const errors = [];
    
    const email = formData.get('email')?.trim();
    const firstName = formData.get('firstName')?.trim();
    const lastName = formData.get('lastName')?.trim();
    const phoneNumber = formData.get('phoneNumber')?.trim();
    const gender = formData.get('gender');
    const role = formData.get('role');
    const province = formData.get('province');
    const municipality = formData.get('municipality');
    const barangay = formData.get('barangay');
    const street = formData.get('street')?.trim();
    
    if (!email || !isValidEmail(email)) errors.push('Valid email is required');
    if (!firstName) errors.push('First name is required');
    if (!lastName) errors.push('Last name is required');
    if (!phoneNumber) errors.push('Phone number is required');
    if (!gender) errors.push('Gender is required');
    if (!role) errors.push('Role is required');
    if (!province || !municipality || !barangay || !street) {
        errors.push('Complete address is required');
    }
    
    return errors;
}

function isValidEmail(email) {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return emailRegex.test(email);
}

        function generateCSV(data) {
            const headers = ['ID', 'Name', 'Contact', 'Email', 'Type', 'Specialization', 'Status', 'Location', 'Emergency Contact'];
            const csvRows = [headers.join(',')];
            
            data.forEach(rescuer => {
                const firstName = rescuer.firstName || extractNameFromEmail(rescuer.email, 'first') || 'Unknown';
                const lastName = rescuer.lastName || extractNameFromEmail(rescuer.email, 'last') || 'User';
                const fullName = `${firstName} ${lastName}`;
                
                const row = [
                    rescuer.id || '',
                    `"${fullName}"`,
                    rescuer.contact || '',
                    rescuer.email || '',
                    rescuer.type || '',
                    `"${rescuer.specialization || ''}"`,
                    rescuer.status || '',
                    `"${rescuer.location || ''}"`,
                    rescuer.emergencyContact || ''
                ];
                csvRows.push(row.join(','));
            });
            
            return csvRows.join('\n');
        }

        // Utility Functions
        function extractNameFromEmail(email, part = 'first') {
            if (!email) return '';
            
            try {
                const username = email.split('@')[0];
                const cleanUsername = username.replace(/[._-]/g, ' ');
                const nameParts = cleanUsername.split(' ').filter(part => part.length > 0);
                
                if (part === 'first' && nameParts.length > 0) {
                    return nameParts[0].charAt(0).toUpperCase() + nameParts[0].slice(1).toLowerCase();
                } else if (part === 'last' && nameParts.length > 1) {
                    return nameParts[nameParts.length - 1].charAt(0).toUpperCase() + nameParts[nameParts.length - 1].slice(1).toLowerCase();
                }
                
                return nameParts[0] ? nameParts[0].charAt(0).toUpperCase() + nameParts[0].slice(1).toLowerCase() : '';
            } catch (error) {
                return '';
            }
        }

        function debounce(func, wait) {
            let timeout;
            return function executedFunction(...args) {
                const later = () => {
                    clearTimeout(timeout);
                    func(...args);
                };
                clearTimeout(timeout);
                timeout = setTimeout(later, wait);
            };
        }
